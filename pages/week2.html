Tutorial: Descriptive Statistics in R

Measures of Central Tendency

# Sample data: wheat yields (tons per hectare)
wheat_yields <- c(3.2, 3.8, 4.1, 3.9, 4.5, 3.7, 4.2, 3.6, 4.0, 3.8)

# Mean
mean(wheat_yields)

# Median
median(wheat_yields)

# Mode (custom function)
get_mode <- function(x) {
  ux <- unique(x)
  ux[which.max(tabulate(match(x, ux)))]
}
get_mode(wheat_yields)

Measures of Variability

# Range
range(wheat_yields)
max(wheat_yields) - min(wheat_yields)

# Variance
var(wheat_yields)

# Standard deviation
sd(wheat_yields)

# Coefficient of variation
cv <- (sd(wheat_yields) / mean(wheat_yields)) * 100
cv

Data Visualization

# Histogram
hist(wheat_yields, 
     main = "Distribution of Wheat Yields",
     xlab = "Yield (tons/ha)",
     col = "lightblue")

# Boxplot
boxplot(wheat_yields,
        main = "Wheat Yield Boxplot",
        ylab = "Yield (tons/ha)")

# Using ggplot2 (install if needed)
library(ggplot2)
data <- data.frame(yields = wheat_yields)

ggplot(data, aes(x = yields)) +
  geom_histogram(bins = 5, fill = "lightblue", color = "black") +
  labs(title = "Wheat Yield Distribution",
       x = "Yield (tons/ha)",
       y = "Frequency")

Class Activity: Soil pH Analysis

Activity 2.1: Comparing Soil pH Across Fields

Analyze soil pH measurements from three different agricultural fields.

Dataset:

# Soil pH data from three fields
field_A <- c(6.2, 6.5, 6.3, 6.7, 6.4, 6.6, 6.1, 6.8, 6.3, 6.5)
field_B <- c(7.1, 7.3, 7.0, 7.2, 7.4, 7.1, 7.3, 7.2, 7.0, 7.1)
field_C <- c(5.8, 6.0, 5.9, 6.1, 5.7, 6.2, 5.8, 6.0, 5.9, 6.1)

Tasks:

  1. Calculate descriptive statistics for each field
  2. Create side-by-side boxplots
  3. Determine which field has the most consistent pH levels
  4. Identify any potential outliers

Solution Framework:

# Combine data
ph_data <- data.frame(
  pH = c(field_A, field_B, field_C),
  Field = rep(c("A", "B", "C"), each = 10)
)

# Summary statistics by field
aggregate(pH ~ Field, data = ph_data, FUN = function(x) {
  c(mean = mean(x), sd = sd(x), cv = sd(x)/mean(x)*100)
})

# Boxplot comparison
boxplot(pH ~ Field, data = ph_data,
        main = "Soil pH by Field",
        ylab = "pH Level",
        col = c("lightcoral", "lightblue", "lightgreen"))
